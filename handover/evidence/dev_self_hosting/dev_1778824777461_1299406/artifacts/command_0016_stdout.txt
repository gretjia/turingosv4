+fn real5_role_turn_outcome(
+    role: turingosv4::runtime::real5_roles::AgentRole,
+    parsed_tool: Option<&str>,
+    parse_error: Option<&str>,
+    policy_rejection: Option<&str>,
+    market_block_present: bool,
+    market_block_budget_elided: bool,
+) -> turingosv4::runtime::real5_roles::RoleTurnOutcome {
+    use turingosv4::runtime::market_decision_trace::NoTradeReason;
+    use turingosv4::runtime::real5_roles::{NoChallengeReason, NoVerifyReason, RoleTurnOutcome};
+
+    if let Some(err) = parse_error {
+        return match role {
+            turingosv4::runtime::real5_roles::AgentRole::Verifier => {
+                RoleTurnOutcome::NoVerify(NoVerifyReason {
+                    agent_id: turingosv4::state::q_state::AgentId(String::new()),
+                    reason: format!("parse_failed: {err}"),
+                })
+            }
+            turingosv4::runtime::real5_roles::AgentRole::Challenger => {
+                RoleTurnOutcome::NoChallenge(NoChallengeReason {
+                    agent_id: turingosv4::state::q_state::AgentId(String::new()),
+                    reason: format!("parse_failed: {err}"),
+                })
+            }
+            _ => RoleTurnOutcome::ParseFailed {
+                public_summary: format!("parse_failed: {err}"),
+            },
+        };
+    }
+    if let Some(summary) = policy_rejection {
+        return RoleTurnOutcome::PolicyRejected {
+            public_summary: summary.to_string(),
+        };
+    }
+
+    match role {
+        turingosv4::runtime::real5_roles::AgentRole::Solver => RoleTurnOutcome::SubmitProof {
+            tx_kind: match parsed_tool {
+                Some("complete") | Some("step") | Some("append") => "WorkTx",
+                _ => "Abstain",
+            }
+            .to_string(),
+        },
+        turingosv4::runtime::real5_roles::AgentRole::Trader => {
+            if parsed_tool == Some("invest") {
+                RoleTurnOutcome::MarketDecision {
+                    tx_kind: "MarketDecisionTrace".to_string(),
+                    public_summary: "invest action emitted; MarketDecisionTrace anchors submitted/no_trade result".to_string(),
+                }
+            } else {
+                let reason = if market_block_present {
+                    NoTradeReason::NoPerceivedEdge
+                } else if market_block_budget_elided {
+                    NoTradeReason::PromptBudgetExceeded
+                } else {
+                    NoTradeReason::NoPool
+                };
+                RoleTurnOutcome::NoTrade {
+                    reason,
+                    public_summary: format!(
+                        "trader role turn emitted {:?}; no invest action",
+                        parsed_tool.unwrap_or("no_parse")
+                    ),
+                }
+            }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::Verifier => {
+            if parsed_tool == Some("verify_peer") {
+                RoleTurnOutcome::VerifyTx {
+                    target_work_tx: None,
+                }
+            } else {
+                RoleTurnOutcome::NoVerify(NoVerifyReason {
+                    agent_id: turingosv4::state::q_state::AgentId(String::new()),
+                    reason: format!(
+                        "verifier emitted {:?}, not verify_peer",
+                        parsed_tool.unwrap_or("no_parse")
+                    ),
+                })
+            }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::Challenger => {
+            if parsed_tool == Some("challenge") {
+                RoleTurnOutcome::ChallengeTx {
+                    target_work_tx: None,
+                }
+            } else {
+                RoleTurnOutcome::NoChallenge(NoChallengeReason {
+                    agent_id: turingosv4::state::q_state::AgentId(String::new()),
+                    reason: format!(
+                        "challenger emitted {:?}, not challenge",
+                        parsed_tool.unwrap_or("no_parse")
+                    ),
+                })
+            }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::MarketMaker => {
+            if parsed_tool == Some("provide_liquidity") {
+                RoleTurnOutcome::MarketDecision {
+                    tx_kind: "MarketSeedTx".to_string(),
+                    public_summary: "market maker liquidity decision emitted".to_string(),
+                }
+            } else {
+                RoleTurnOutcome::Abstain {
+                    reason: "no liquidity action emitted".to_string(),
+                }
+            }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::Architect => {
+            RoleTurnOutcome::ToolProposal { proposal_id: None }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::Veto => {
+            RoleTurnOutcome::VetoDecision { proposal_id: None }
+        }
+        turingosv4::runtime::real5_roles::AgentRole::Observer => RoleTurnOutcome::Abstain {
+            reason: "observer role does not emit production action".to_string(),
+        },
+    }
+}

+fn real5_write_role_turn_trace(
+    bundle: &turingosv4::runtime::ChaintapeBundle,
+    run_id: &str,
+    agent_id: &str,
+    role: turingosv4::runtime::real5_roles::AgentRole,
+    prompt_capsule_cid: turingosv4::bottom_white::cas::schema::Cid,
+    parsed_tool: Option<&str>,
+    parse_error: Option<&str>,
+    policy_rejection: Option<&str>,
+    market_block_present: bool,
+    market_block_budget_elided: bool,
+    logical_t: u64,
+) -> Result<(), String> {
+    use turingosv4::runtime::real5_roles::{
+        write_role_turn_trace_to_cas, NoChallengeReason, NoVerifyReason, RoleTurnOutcome,
+        RoleTurnTrace,
+    };
+
+    let mut outcome = real5_role_turn_outcome(
+        role,
+        parsed_tool,
+        parse_error,
+        policy_rejection,
+        market_block_present,
+        market_block_budget_elided,
+    );
+    match &mut outcome {
+        RoleTurnOutcome::NoVerify(NoVerifyReason {
+            agent_id: trace_agent_id,
+            ..
+        })
+        | RoleTurnOutcome::NoChallenge(NoChallengeReason {
+            agent_id: trace_agent_id,
+            ..
+        }) => {
+            *trace_agent_id = turingosv4::state::q_state::AgentId(agent_id.to_string());
+        }
+        _ => {}
+    }
+
+    let trace = RoleTurnTrace::new(
+        turingosv4::state::q_state::AgentId(agent_id.to_string()),
+        role,
+        turingosv4::state::q_state::TaskId(format!("task-{run_id}")),
+        prompt_capsule_cid,
+        parsed_tool.map(str::to_string),
+        outcome,
+    );
+    let mut cas_store = turingosv4::bottom_white::cas::store::CasStore::open(&bundle.cas_path)
+        .map_err(|e| format!("REAL-5 role turn CAS open: {e}"))?;
+    write_role_turn_trace_to_cas(
+        &mut cas_store,
+        &trace,
+        &format!("{}-{}", agent_id, logical_t),
+        logical_t,
+    )
+    .map_err(|e| format!("REAL-5 role turn CAS write: {e}"))?;
+    Ok(())
